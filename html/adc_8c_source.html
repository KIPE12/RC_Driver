197void AdcProcess(void)
198{
199 ADC1_Result[0] = (adc1Val[0] - ADC1_Offset[0]) / 81.9f;
200 ADC1_Result[1] = (adc1Val[1] - ADC1_Offset[1]) / 81.9f;
201 ADC1_Result[2] = (adc1Val[2] - ADC1_Offset[2]) / 81.9f;
202 ADC1_Result[3] = adc1Val[3] / 203.4f;
203
204 INV.Ia = ADCgain[0] * ADC1_Result[0] * scale_comp;
205 INV.Ib = ADCgain[1] * ADC1_Result[1] * scale_comp;
206 INV.Ic = ADCgain[2] * ADC1_Result[2] * scale_comp;
207 INV.Vdc = ADCgain[3] * ADC1_Result[3] * scale_comp;
208
209 const float alpha = 0.999f;
210 INV.Vdc_control = INV.Vdc * (1.0f - alpha) + alpha * INV.Vdc_control;
211 INV.Vdc = INV.Vdc_control;
212 INV.INV_Vdc = 1.0f / MAX(INV.Vdc_control, 1.f);
213
214 if (store_flag == 1) {
215 if (store_cnt < 3000) Ia_arr[store_cnt++] = INV.Ia;
216 else store_flag = 0;
217 }
218}

230void Offset(void)
231{
232 static unsigned int AdDummyCnt = 0;
233

235 AdDummyCnt++;

237 AdOffCalcCnt++;

240 ADC1_Offset_sum[1] += (unsigned long)adc1Val[1];
241 ADC1_Offset_sum[2] += (unsigned long)adc1Val[2];
242
243 if (AdOffCalcCnt == 5000) {
244 ADC1_Offset[0] = (float)ADC1_Offset_sum[0] / 5000.f;
245 ADC1_Offset[1] = (float)ADC1_Offset_sum[1] / 5000.f;
246 ADC1_Offset[2] = (float)ADC1_Offset_sum[2] / 5000.f;
247 AdInitFlag = 1;
248 }
