197void SystemCoreClockUpdate(void)
198{
199 uint32_t tmp, pllvco, pllr, pllsource, pllm;
200
201 /* Get SYSCLK source -------------------------------------------------------*/
202 switch (RCC->CFGR & RCC_CFGR_SWS)
203 {
204 case 0x04: /* HSI used as system clock source */
205 SystemCoreClock = HSI_VALUE;
206 break;
207
208 case 0x08: /* HSE used as system clock source */
209 SystemCoreClock = HSE_VALUE;
210 break;
211
212 case 0x0C: /* PLL used as system clock source */
213 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLLM) * PLLN
214 SYSCLK = PLL_VCO / PLLR
215 */
216 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC);
217 pllm = ((RCC->PLLCFGR & RCC_PLLCFGR_PLLM) >> 4) + 1U ;
218 if (pllsource == 0x02UL) /* HSI used as PLL clock source */
219 {
220 pllvco = (HSI_VALUE / pllm);
221 }
222 else /* HSE used as PLL clock source */
223 {
224 pllvco = (HSE_VALUE / pllm);
225 }
226 pllvco = pllvco * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 8);
227 pllr = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLR) >> 25) + 1U) * 2U;
228 SystemCoreClock = pllvco/pllr;
229 break;
230
231 default:
232 break;
233 }
234 /* Compute HCLK clock frequency --------------------------------------------*/
235 /* Get HCLK prescaler */
236 tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
237 /* HCLK clock frequency */
238 SystemCoreClock >>= tmp;
239}
